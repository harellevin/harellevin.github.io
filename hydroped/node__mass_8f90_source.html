00048   use sizes
00049   use node_data
00050   use element_data
00051   implicit none
00052 
00053   integer::num,i
00054   real(kind=8)::xn(4),yn(4),zn(4)
00055   real(kind=8)::vol,vola,volmin,volmax,elmass
00056   real(kind=8)::v1,v2,v3,v4
00057 
00058   mass=0.0_8 !zero mass array
00059 
00060   vola = 0.0_8
00061   volmin = 999999999999.0_8
00062   volmax = 0.0_8
00063 
00064   ! loop for all elements
00065   do num = 1,ne
00066     ! coord 4 element nodes
00067     do i = 1,4
00068       xn(i) = cord(nop(num,i),1)
00069       yn(i) = cord(nop(num,i),2)
00070       zn(i) = cord(nop(num,i),3)
00071     end do
00072 
00073     !element volume  (G = 6 * volume )
00074     !| 1 x1 y1 z1 |
00075     !| 1 x2 y2 z2 |
00076     !| 1 x3 y3 z3 |
00077     !| 1 x4 y4 z4 |
00078     v1= xn(2) * (yn(3)*zn(4)-zn(3)*yn(4)) &
00079       - xn(3) * (yn(2)*zn(4)-zn(2)*yn(4)) &
00080       + xn(4) * (yn(2)*zn(3)-zn(2)*yn(3)) 
00081 
00082     v2= xn(1) * (yn(3)*zn(4)-zn(3)*yn(4)) &
00083       - xn(3) * (yn(1)*zn(4)-zn(1)*yn(4)) &
00084       + xn(4) * (yn(1)*zn(3)-zn(1)*yn(3))
00085 
00086     v3= xn(1) * (yn(2)*zn(4)-zn(2)*yn(4)) &
00087       - xn(2) * (yn(1)*zn(4)-zn(1)*yn(4)) &
00088       + xn(4) * (yn(1)*zn(2)-zn(1)*yn(2))
00089 
00090     v4= xn(1) * (yn(2)*zn(3)-zn(2)*yn(3)) &
00091       - xn(2) * (yn(1)*zn(3)-zn(1)*yn(3)) &
00092       + xn(3) * (yn(1)*zn(2)-zn(1)*yn(2))
00093 
00094     vol = v1 - v2 + v3 - v4
00095     vol = abs(vol)/6.0_8
00096 
00097     vola = vola + vol
00098 
00099     if ( volmin .ge. vol ) volmin = vol
00100     if ( volmax .le. vol ) volmax = vol
00101 
00102     el_vol(num) = vol
00103     elmass = dens(num) * vol
00104     do i = 1,4
00105       mass(nop(num,i)) = mass(nop(num,i)) + 0.25_8*elmass
00106     end do
00107   end do
00108 
00109   vola = vola / ne
00110   write(6,*)' Average element volume is ',vola
00111   write(6,*)' Volume ranges from ',volmin,'  to ',volmax
