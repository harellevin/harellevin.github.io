00028 !  Move grid along veocity field (Lagrangian) and new total strain tensor  
00038 subroutine move_grid(dt)
00039   
00040   use sizes
00041   use node_data
00042   use element_data
00043 
00044   implicit none
00045   
00046   integer:: ii,j,i,n,nn
00047   real(kind=8)::dt,de(6),dr(4,3)
00048   
00049   ! moving of mesh
00050   !$OMP PARALLEL
00051   !$OMP DO PRIVATE(i,j)
00052   do i=1,np
00053     do j=1,3
00054       cord(i,j) = cord(i,j) + vel(i,j)*dt
00055       disp(i,j) = disp(i,j) + vel(i,j)*dt
00056       dspt(i,j) = dspt(i,j) + vel(i,j)*dt
00057     end do
00058   end do
00059   !$OMP END DO
00060   !$OMP END PARALLEL 
00061 
00062   ! New elastic strain tensor
00063   !$OMP PARALLEL
00064   !$OMP DO PRIVATE(dr,n,ii,de,j,nn)
00065   do n = 1,ne
00066  ! derivation of basic functions
00067     call derivation(dr,n)
00068     
00069     ! Strain rate
00070     de = 0.0_8
00071     do ii=1,4
00072       nn = nop(n,ii)
00073       de(1)=de(1) + vel(nn,1)*dr(ii,1)
00074       de(2)=de(2) + vel(nn,2)*dr(ii,2)
00075       de(3)=de(3) + vel(nn,3)*dr(ii,3)
00076       de(4)=de(4) + 0.5_8*(vel(nn,1)*dr(ii,2)+vel(nn,2)*dr(ii,1))
00077       de(5)=de(5) + 0.5_8*(vel(nn,1)*dr(ii,3)+vel(nn,3)*dr(ii,1))
00078       de(6)=de(6) + 0.5_8*(vel(nn,2)*dr(ii,3)+vel(nn,3)*dr(ii,2))
00079     end do
00080 
00081  ! New total strains
00082     do j=1,6
00083       strain(j,n) = strain(j,n) + de(j)*dt
00084     end do
00085   end do
00086   !$OMP END DO
00087   !$OMP END PARALLEL
00088   
00089 return
00090 end subroutine move_grid
