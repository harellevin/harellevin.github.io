00042   logical::event
00043   integer::n,j,i
00044   real(kind=8)::dt,vvv,s(6)
00045   
00046 ! Initialization 
00047   event=.false.
00048   force=0.0_8
00049   balance=0.0_8
00050  
00051 ! loop through all elements
00052   !$OMP PARALLEL

00073 ! Stress for next step

00100 subroutine force_balance(n)
00101   use sizes
00102   use node_data
00103   use element_data
00104   implicit none
00105 
00106   logical::event,space
00107   integer::n,nside,nn,j,ii,n1,n2,n3
00108   real(kind=8)::x1,x2,y1,y2,z1,z2,xnorm,ynorm,znorm
00109   real(kind=8)::fs(4,3),f(4,3)
00110 
00111   ! Calculation forces (4 sides, one element)
00112   ! On the sides (fs(i,j), i-side, j-dimension) fi = sij*nj
00113   do nside = 1,4
00114     ! Three nodes of the side
00115     n1 = nop(n,side(1,nside))
00116     n2 = nop(n,side(2,nside))
00117     n3 = nop(n,side(3,nside))
00118     ! Two vectors of the side
00119     x1 = cord(n2,1) - cord(n1,1)
00120     y1 = cord(n2,2) - cord(n1,2)
00121     z1 = cord(n2,3) - cord(n1,3)
00122 
00123     x2 = cord(n3,1) - cord(n1,1)
00124     y2 = cord(n3,2) - cord(n1,2)
00125     z2 = cord(n3,3) - cord(n1,3)
00126 
00127     ! Normal vector to the side (NORM = X1 X X2 / 2.)
00128     xnorm = 0.5_8*(y1*z2 - z1*y2)
00129     ynorm = 0.5_8*(z1*x2 - x1*z2)
00130     znorm = 0.5_8*(x1*y2 - y1*x2)
00131 
00132     fs(nside,1) = stress(1,n)*xnorm + stress(4,n)*ynorm + stress(5,n)*znorm
00133     fs(nside,2) = stress(4,n)*xnorm + stress(2,n)*ynorm + stress(6,n)*znorm
00134     fs(nside,3) = stress(5,n)*xnorm + stress(6,n)*ynorm + stress(3,n)*znorm
00135  end do
00136 
00137 
00138   ! In the nodes (f(i,j), i-node,j-dimension) 
00139   ! each node belongs to three sides (see "side" above)
00140   do j=1,3
00141     f(1,j) = (fs(1,j)+fs(2,j)+fs(4,j))/3.0_8
00142     f(2,j) = (fs(1,j)+fs(3,j)+fs(4,j))/3.0_8
00143     f(3,j) = (fs(2,j)+fs(3,j)+fs(4,j))/3.0_8
00144     f(4,j) = (fs(1,j)+fs(2,j)+fs(3,j))/3.0_8
00145   end do
00146 
00147   ! Insert forces in whole system:
00148   do ii=1,4
00149     nn = nop(n,ii)
00150     do j=1,3
00151       force(nn,j) = force(nn,j) - f(ii,j)
00152       balance(nn,j) = balance(nn,j) + abs(f(ii,j))
00153     end do
00154   end do
00155    
00156   return
00157 end subroutine force_balance

00160 subroutine n_force_balance(dt)
00161   use sizes
00162   use node_data
00163   use boundary_node_data
00164   use element_data
00165   implicit none
00166 
00167   logical:: space
00168   integer:: j,nb,i,ii,i3,nrd,node,nn,nd
00169   real(kind=8):: fs(4,3),f(4,3),fv,vvs,fff,bbb,vvv,vmax,dt
00170   
00171   ! Including volumetric force
00172   !$OMP PARALLEL
00173   !$OMP DO PRIVATE(i,i3,fv)
00174   do i = 1, np
00175     do i3 = 1,3
00176       fv = mass(i)*g(i3)
00177       force(i,i3)   = force(i,i3) - fv
00178       balance(i,i3) = balance(i,i3) + abs(fv)
00179     end do
00180   end do
00181   !$OMP END DO
00182 
00183   ! BOUNDARY CONDITION IN STRESSES
00184   !$OMP DO PRIVATE(j,nb,nn)
00185   do j=1,3
00186     do nb = 1,nbp
00187       if( force_code(nb,j) )then
00188         nn = numbn( nb )
00189         force(nn,j)   = force(nn,j)   + value(nb,j)
00190         balance(nn,j) = balance(nn,j) + abs (value(nb,j))
00191       end if
00192     end do
00193   end do
00194   !$OMP END DO  
00195   
00196   !$OMP DO PRIVATE(i,j,vvs,fff)
00197   ! Damping
00198   do i=1,np
00199     do j=1,3
00200       vvs = vel(i,j) + dt*force(i,j)/(mass(i)*den_scale)
00201       if ( abs(vvs) .gt. 1.0e-20_8) then
00202         fff = abs(force(i,j))
00203         force(i,j) = force(i,j) - demf*sign(fff,vvs)
00204         vel(i,j)=vel(i,j) + dt*force(i,j)/(mass(i)*den_scale)
00205       end if
00206     end do
00207   end do
00208   !$OMP END DO
00209   
00210   ! Boundary condition in velocities:
00211   !$OMP DO PRIVATE(i,j,node)
00212   do i=1,nbp 
00213     do j=1,3 
00214       if( vel_code(i,j) )then
00215         node = numbn(i)
00216         vel(node,j)= value(i,j)
00217         force(node,j)= 0.0_8
00218         balance(node,j)= 0.0_8
00219       end if
00220     end do
00221   end do
00222   !$OMP END DO
00223   
00224   ! Put velocity & balance to zero if node is in open space
00225   !$OMP DO PRIVATE(i,j,space,nrd,nd)
00226   do i =1,np
00227     nrd = nos(i,1)
00228     space = .true.
00229     do nd = 2,nrd+1
00230       space = space .and. (flag( nos(i,nd) ) .ne. 0 )
00231     end do
00232 
00233     if(space)then       ! open space
00234       do j = 1,3
00235         vel(i,j) = 0.0_8
00236         force(i,j) = 0.0_8
00237         balance(i,j) = 0.0_8
00238       end do
00239     end if
00240   end do
00241   !$OMP END DO
00242   !$OMP END PARALLEL
00243     
00244   return
00245 end subroutine n_force_balance
