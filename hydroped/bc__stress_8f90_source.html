00037   integer::n,j,i,nb,nn,i3
00038   real(kind=8)::fv
00039 
00040 
00041 !  Initialization 
00042    force=0.0_8
00043 
00044   do n=1,ne
00045 ! calculate and transfer element forces onto nodes
00046     call force_balance(n)
00047   end do
00048 
00049 !----- BOUNDARY CONDITION in STRESSES for boundary points

00051     do j=1,3
00052         nn = numbn(nb)
00053 
00054 !-------- Including volumetric force

00057           force(nn,i3)   = force(nn,i3) - fv
00058        end do
00059    
00060    if(force_code(nb,j)) value(nb,j) = -force(nn,j)
00061 
00062     end do
00063   end do
00064 
00065   return
