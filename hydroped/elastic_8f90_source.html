00028 ! Elastic stress-strain test for connectivity
00029 !   flg = 0    -  stable
00030 !   flg = 1    -  1-st type of failure
00031 !   flg = 2    -  2-nd type of failure
00032 !   flg = 3    -  open space
00033 
00058 subroutine elastic(n,event,s)
00059 use sizes
00060 use element_data
00061 implicit none
00062 
00063 integer:: n,j
00064 logical:: event
00065 real(kind=8):: rl,rm,rg,s(6)
00066 real(kind=8):: si2,a,b,xmin,p,q,d,cs
00067 
00068  if( flag(n).eq.3 ) then    ! open space 
00069     do j=1,6
00070       s(j) = 0.0_8
00071     end do
00072  else ! normal cell
00073     ! invariants of the strain tensor
00074     call strain_invariants(n,si2)
00075 
00076     ! stress calculation, zero if loss convexity
00077     ! effective moduli
00078     rl  = lambda(n)
00079     rm  = mu(n) + alpha(n)*ksi0(n)*gr(n)
00080     rg  = alpha(n)*gr(n)
00081  
00082    cs = ksi(n)
00083    
00084     a =2.0_8*rm-rg*ksi(n)
00085     p =-(4.0_8*rm+3.0_8*rl-3.0_8*rg*cs)
00086     q =a**2 + a*(3.0_8*rl-rg*cs) + (rl*rg*cs-rg*rg)*(3.0_8-cs*cs)
00087     d = 0.25_8*p**2 - q
00088        
00089     if ( d .le. 0.0_8 ) then 
00090       write(6,*)' Discreminant < 0; I do not know how to continue ',d
00091       stop
00092     end if
00093     xmin = -0.5_8*p - sqrt(d)
00094     if ( xmin .le. 0.0_8 ) then
00095       flag(n) = 2
00096       event=.true.
00097       do j=1,6
00098         s(j) = 0.0_8
00099       end do
00100     else if ( a .le. 0.0_8 ) then
00101       flag(n) = 1
00102       event=.true.
00103       do j=1,6
00104         s(j) = 0.0_8
00105       end do
00106     else
00107       b=rl*i1(n) - rg*si2
00108       do j=1,3
00109         s(j) = b + a*str_e(j,n)
00110         s(j+3) =   a*str_e(j+3,n)
00111       end do
00112     end if
00113  end if
00114 
00115   return 
00116 end subroutine elastic
